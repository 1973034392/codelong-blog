            totalRequests++;
            if (!success) failures++;
            
            double failureRate = (double) failures / totalRequests;
            if (totalRequests >= minRequests && failureRate >= failureThreshold) {
                open();
            }
        }

                if (System.currentTimeMillis() - openedAt > timeoutMs) {
                    // å°è¯•åŠå¼€çŠ¶æ€

                    totalRequests = 0;
                    failures = 0;
                    return false;
                }
                return true;
            }
            return false;
        }
        
        private void open() {
            openedAt = System.currentTimeMillis();
        }
    }
